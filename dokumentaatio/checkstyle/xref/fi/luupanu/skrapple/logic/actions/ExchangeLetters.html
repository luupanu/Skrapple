6   package fi.luupanu.skrapple.logic.actions;
7   
8   import fi.luupanu.skrapple.domain.Game;
9   import fi.luupanu.skrapple.domain.Letter;
10  import java.util.ArrayList;
11  import java.util.List;
12  
13  /**
14   * A player can choose to use her turn to exchange one or more of her letters.
15   *
16   * @author panu
17   */
18  public class ExchangeLetters extends Action {
19  
20      private final List<Integer> indexes;
21  
22      public ExchangeLetters(Game game, List<Integer> indexes) {
23          super(game);
24          this.indexes = indexes;
25      }
26  
27      @Override
28      public void perform(Game game) {
29          List<Letter> letters = new ArrayList<>(7);
30          for (Integer i : indexes) {
31              letters.add(game.getCurrentPlayer().getPlayerRack().takeLetter(i));
32          }
33          game.getCurrentPlayer().getPlayerRack().refillRack(game.getLetterBag());
34          for (Letter let : letters) {
35              game.getLetterBag().placeLetterInBag(let);
36          }
37      }
38  
39  }
