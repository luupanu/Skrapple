6   package fi.luupanu.skrapple.logic.actions;
7   
8   import fi.luupanu.skrapple.domain.Game;
9   import fi.luupanu.skrapple.constants.SkrappleGameState;
10  
11  /**
12   * A player can choose to resign the game during her turn.
13   *
14   * @author panu
15   */
16  public class Resign extends Action {
17  
18      /**
19       * Creates a new action Resign.
20       *
21       * @param game the game that is being played
22       */
23      public Resign(Game game) {
24          super(game);
25      }
26  
27      /**
28       * Checks which player's turn it is, and changes the game state accordingly.
29       *
30       * @param game the game being played
31       */
32      @Override
33      public void perform(Game game) {
34          if (game.getTurn()) {
35              game.setGameState(SkrappleGameState.PLAYER_1_RESIGNED);
36          } else {
37              game.setGameState(SkrappleGameState.PLAYER_2_RESIGNED);
38          }
39      }
40  }
