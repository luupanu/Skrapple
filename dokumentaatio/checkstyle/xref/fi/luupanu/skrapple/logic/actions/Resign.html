6   package fi.luupanu.skrapple.logic.actions;
7   
8   import fi.luupanu.skrapple.constants.ErrorMessage;
9   import fi.luupanu.skrapple.constants.GameState;
10  import fi.luupanu.skrapple.domain.Game;
11  import fi.luupanu.skrapple.domain.Player;
12  
13  /**
14   * A player can choose to resign the game during her turn.
15   *
16   * @author panu
17   */
18  public class Resign extends GameAction {
19  
20      /**
21       * Creates a new action Resign.
22       *
23       * @param game the game that is being played
24       */
25      public Resign(Game game) {
26          super(game);
27      }
28  
29      /**
30       * Checks which player's turn it is, and changes the game state accordingly.
31       *
32       * @param game the game being played
33       * @return true
34       */
35      @Override
36      public ErrorMessage perform(Game game) {
37          if (game.getGameState() == GameState.GAMEOVER) {
38              return ErrorMessage.GAME_IS_OVER;
39          }
40          game.getCurrentPlayer().resign();
41          if (gameIsOver(game)) {
42              game.setGameState(GameState.GAMEOVER);
43          }     
44          return ErrorMessage.NO_ERRORS;
45      }
46  
47      private boolean gameIsOver(Game game) {
48          int resignedPlayers = 0;
49          for (Player p : game.getPlayerList()) {
50              if (p.isResigned()) {
51                  resignedPlayers++;
52              }
53          }
54          return resignedPlayers == game.getPlayerList().size() - 1;
55      }
56  }
