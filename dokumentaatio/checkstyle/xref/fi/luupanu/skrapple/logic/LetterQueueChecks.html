6   package fi.luupanu.skrapple.logic;
7   
8   import fi.luupanu.skrapple.domain.Board;
9   import fi.luupanu.skrapple.domain.Coord;
10  import fi.luupanu.skrapple.domain.Letter;
11  import fi.luupanu.skrapple.domain.Square;
12  import java.util.List;
13  
14  /**
15   * This class houses checks that are performed to all letters before they are
16   * added to the LetterQueue.
17   *
18   * @author panu
19   */
20  public class LetterQueueChecks {
21  
22      private final LetterQueue q;
23      private final List<Letter> list;
24  
25      /**
26       * Creates the class that checks all the letters to be added to the
27       * LetterQueue.
28       *
29       * @param queue the letter queue used by the game
30       */
31      public LetterQueueChecks(LetterQueue queue) {
32          this.q = queue;
33          this.list = queue.getLetterQueue();
34      }
35  
36      /**
37       * Does a variety of checks to a letter before it is added to the queue. The
38       * checks are used to guide the player in making a valid move. The body of
39       * the method contains more explanations for all the checks.
40       *
41       * @param let the letter to be added
42       * @param c the coordinates for the letter
43       * @param board the game board
44       * @return true if the letter is allowed to be added to the queue
45       */
46      public boolean letterCanBeAddedToQueue(Letter let, Coord c, Board board) {
47          int x = c.getX();
48          int y = c.getY();
49  
50          /*  preliminary checks: is this a valid coordinate, is this a valid
51              letter, does the queue already have a letter at the coordinate? */
52          if (!q.isValidCoordinate(x, y) || q.hasCoord(x, y) || let == null) {
53              return false;
54          }
55  
56          // check that the board doesn't already have a letter at the coordinate
57          if (board.getSquare(x, y).hasLetter()) {
58              return false;
59          }
60  
61          // the first letter of the first word must be at x = 7 OR y = 7
62          if (board.hasNoLetters() && list.isEmpty()) {
63              return x == 7 || y == 7;
64          }
65  
66          // check that the first letter of the queue is placed correctly
67          if (list.isEmpty()) {
68              return checkFirstLetterCorrectPlacement(c, board);
69          }
70  
71          // check that the second letter of the queue is placed correctly
72          if (list.size() == 1) {
73              return checkSecondLetterCorrectPlacement(c, board);
74          }
75  
76          // now that we have a direction for the word, we can check the rest
77          return checkRemainingLettersCorrectPlacement(c, board, q.getDirection());
78      }
79  
80      private boolean checkFirstLetterCorrectPlacement(Coord c, Board board) {
81          /*  First letter of the word must be on the same row or column as an
82              existing letter on the board, or on the neighbouring row/column */
83          return rowOrNeighbouringRowHasLetter(c.getY(), board)
84                  || columnOrNeighbouringColumnHasLetter(c.getX(), board);
85      }
86  
87      private boolean checkSecondLetterCorrectPlacement(Coord c, Board board) {
88          /*  Second letter of the word must be on the same row or column as the
89              first letter. The second letter gives us the direction for all the 
90              rest of the letters (a player can either place a word vertically or
91              horizontally on the board, not both at the same time) */
92          for (Letter let : list) {
93              Coord firstLetterCoord = let.getCoord();
94              if (firstLetterCoord.getX() == c.getX() && firstLetterCoord.getY() == c.getY()) {
95                  return false;
96              } else if (firstLetterCoord.getX() == c.getX()) {
97                  q.setDirection(false); // vertical
98                  return true;
99              } else if (firstLetterCoord.getY() == c.getY()) {
100                 q.setDirection(true); // horizontal
101                 return true;
102             }
103         }
104         return false;
105     }
106 
107     private boolean checkRemainingLettersCorrectPlacement(Coord c, Board board, boolean direction) {
108         for (Letter let : list) {
109             Coord coordinate = let.getCoord();
110             if (c.getY() == coordinate.getY() && c.getX() == coordinate.getX()) {
111                 return false; // don't try to add exact same coordinates
112             }
113             if (direction) { // horizontal
114                 if (c.getY() != coordinate.getY()) {
115                     return false;
116                 }
117             } else if (!direction) { // vertical
118                 if (c.getX() != coordinate.getX()) {
119                     return false;
120                 }
121             }
122         }
123         return true;
124     }
125 
126     private boolean rowOrNeighbouringRowHasLetter(int row, Board board) {
127         Square[][] b = board.getContents();
128 
129         for (int y = row - 1; y <= row + 1; y++) {
130             for (int x = 0; x < 15; x++) {
131                 if (!q.isValidCoordinate(x, y)) {
132                     continue;
133                 }
134                 Square s = b[x][y];
135                 if (s.hasLetter()) {
136                     return true;
137                 }
138             }
139         }
140         return false;
141     }
142 
143     private boolean columnOrNeighbouringColumnHasLetter(int col, Board board) {
144         Square[][] b = board.getContents();
145 
146         for (int y = 0; y < 15; y++) {
147             for (int x = col - 1; x < col + 1; x++) {
148                 if (!q.isValidCoordinate(x, y)) {
149                     continue;
150                 }
151                 Square s = b[x][y];
152                 if (s.hasLetter()) {
153                     return true;
154                 }
155             }
156         }
157         return false;
158     }
