12  public class SkrappleGame {
13  
14      private Player p1;
15      private Player p2;
16      private Board board;
17      private SkrappleGameState state;
18      private boolean whoseTurn;
19  
20      public SkrappleGame(Player p1, Player p2) {
21          this.p1 = p1;
22          this.p2 = p2;
23          createGame();
24      }
25      // TODO: the main game loop
26  
27      public Player play() {
28          // still under construction. Returns the winner of the game
29          while (true) {
30              Turn turn = new Turn();
31              turn.doTurn(whoseTurn);
32              if (state != SkrappleGameState.STATE_PLAYING) {
33                  break;
34              }
35              switchTurn();
36          }
37  
38          subtractRemainingLetters();
39          return declareWinner();
40      }
41  
42      private void createGame() {
43          state = SkrappleGameState.STATE_PLAYING;
44          board = new Board();
45          whoseTurn = true; // true = player one's turn
46      }
47  
48      private void subtractRemainingLetters() {
49          p1.addPoints(-p1.getPlayerRack().getRackPoints());
50          p2.addPoints(-p2.getPlayerRack().getRackPoints());
51      }
52  
53      private Player declareWinner() {
54          if (state == SkrappleGameState.STATE_PLAYER_2_RESIGNED) {
55              return p1;
56          } else if (state == SkrappleGameState.STATE_PLAYER_1_RESIGNED) {
57              return p2;
58          }
59          if (p1.getPlayerPoints() > p2.getPlayerPoints()) {
60              return p1;
61          } else if (p1.getPlayerPoints() < p2.getPlayerPoints()) {
62              return p2;
63          }
64          return null;
65      }
66              
67      public boolean getTurn() {
68          return whoseTurn;
69      }
70      
71      public void switchTurn() {
72          whoseTurn = !getTurn();
73      }
74  
75  }
