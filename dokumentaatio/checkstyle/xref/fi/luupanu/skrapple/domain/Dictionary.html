6   package fi.luupanu.skrapple.domain;
7   
8   import fi.luupanu.skrapple.filemanager.FileReader;
9   import java.io.FileNotFoundException;
10  import java.util.List;
11  import java.util.logging.Level;
12  import java.util.logging.Logger;
13  
14  /**
15   * The Dictionary class houses the word list used to evaluate if a word will be
16   * approved or not.
17   *
18   * @author panu
19   */
20  public class Dictionary {
21  
22      private final List<String> wordlist;
23  
24      public Dictionary(String filename) {
25          wordlist = loadWordList(filename);
26      }
27  
28      private List<String> loadWordList(String filename) {
29          try {
30              return new FileReader().readFile(filename);
31          } catch (FileNotFoundException ex) {
32              Logger.getLogger(Dictionary.class.getName()).log(Level.SEVERE, null, ex);
33              return null;
34          }
35      }
36  
37      public List<String> getWordList() {
38          return wordlist;
39      }
40  
41      public boolean containsWord(String word) {
42          return wordlist.contains(word);
43      }
44  }
