6   package fi.luupanu.skrapple.domain;
7   
8   import fi.luupanu.skrapple.constants.SquareType;
9   import java.util.ArrayList;
10  import java.util.List;
11  
12  /**
13   * Words are made of letters. This class is also used to update the score total
14   * of a word. The score of a word is the sum of the scores of its letters, each
15   * multiplied according to any letter bonus squares newly covered, then finally
16   * multiplied according to any word bonus squares newly covered.
17   * 
18   * @author panu
19   */
20  public class Word {
21  
22      private final List<Letter> word;
23      private int points;
24      private int wordCoefficient;
25  
26      public Word() {
27          word = new ArrayList<>(15);
28          points = 0;
29          wordCoefficient = 1;
30      }
31  
32      public boolean addLetter(Letter let, Board board, boolean isQueueLetter) {
33          if (isQueueLetter && let.getCoord() == null) {
34              return false;
35          }
36          updatePoints(let, board, isQueueLetter);
37          word.add(let);
38          return true;
39      }
40      
41      public int getPoints() {
42          return points * wordCoefficient;
43      }
44      
45      public List<Letter> getWord() {
46          return word;
47      }
48  
49      private void updatePoints(Letter let, Board board, boolean isQueueLetter) {
50          int letterCoefficient = 1;
51          if (isQueueLetter) {
52              int x = let.getCoord().getX();
53              int y = let.getCoord().getY();
54              SquareType t = board.getSquare(x, y).getType();
55              if (t == SquareType.SQUARE_NORMAL) {
56                  letterCoefficient = 1;
57              } else if (t == SquareType.SQUARE_BONUS_LETTER_2X) {
58                  letterCoefficient = 2;
59              } else if (t == SquareType.SQUARE_BONUS_WORD_2X) {
60                  wordCoefficient *= 2;
61              } else if (t == SquareType.SQUARE_BONUS_LETTER_3X) {
62                  letterCoefficient = 3;
63              } else if (t == SquareType.SQUARE_BONUS_WORD_3X) {
64                  wordCoefficient *= 3;
65              }
66          }
67          points += let.getPoints() * letterCoefficient;
68      }
69      
70      @Override
71      public String toString() {
72          StringBuilder sb = new StringBuilder();
73          for (Letter let : word) {
74              sb.append(let.toString());
75          }
76          return sb.toString();
77      }
78  }
