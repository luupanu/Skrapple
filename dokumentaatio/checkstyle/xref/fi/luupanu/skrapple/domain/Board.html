6   package fi.luupanu.skrapple.domain;
7   
8   import fi.luupanu.skrapple.constants.SquareType;
9   
10  /**
11   * Board is the main board used by the game. Board has 15x15 = total 225
12   * Squares, including 164 normal, 24 double letter, 17 double word, 12 triple
13   * letter and 8 triple word bonus squares.
14   *
15   * @author panu
16   */
17  public class Board {
18  
19      private static final int BOARD_MAX_SIZE = 15;
20      private final String[] layout = {
21          "W..l...W...l..W",
22          ".w...L...L...w.",
23          "..w...l.l...w..",
24          "l..w...l...w..l",
25          "....w.....w....",
26          ".L...L...L...L.",
27          "..l...l.l...l..",
28          "W..l...w...l..W",
29          "..l...l.l...l..",
30          ".L...L...L...L.",
31          "....w.....w....",
32          "l..w...l...w..l",
33          "..w...l.l...w..",
34          ".w...L...L...w.",
35          "W..l...W...l..W"};
36  
37      private Square[][] board;
38  
39      /**
40       * Creates the default Skrapple board based on the layout.
41       */
42      public Board() {
43          createDefaultBoard();
44      }
45  
46      public Square[][] getContents() {
47          return board;
48      }
49      
50      public String[] getLayout() {
51          return layout;
52      }
53      
54      public int getBoardMaxSize() {
55          return BOARD_MAX_SIZE;
56      }
57  
58      /**
59       * Returns a square based on coordinates if they are valid.
60       *
61       * @param x the x coordinate
62       * @param y the y coordinate
63       * @return the Square in the coordinates
64       */
65      public Square getSquare(int x, int y) {
66          if (x >= 0 && x < BOARD_MAX_SIZE && y >= 0 && y < BOARD_MAX_SIZE) {
67              return board[y][x];
68          }
69          return null;
70      }
71  
72      /**
73       * This method is useful in evaluating if it is the first word of the game.
74       *
75       * @return true if board has no letters
76       */
77      public boolean hasNoLetters() {
78          for (int y = 0; y < board.length; y++) {
79              for (int x = 0; x < board[y].length; x++) {
80                  if (getSquare(x, y).hasLetter()) {
81                      return false;
82                  }
83              }
84          }
85          return true;
86      }
87  
88      private void createDefaultBoard() {
89          board = new Square[BOARD_MAX_SIZE][BOARD_MAX_SIZE];
90          for (int y = 0; y < board.length; y++) {
91              String row = layout[y];
92              for (int x = 0; x < board[y].length; x++) {
93                  SquareType type;
94                  char c = row.charAt(x);
95                  if (c == '.') {
96                      type = SquareType.NORMAL;
97                  } else if (c == 'l') {
98                      type = SquareType.BONUS_LETTER_2X;
99                  } else if (c == 'w') {
100                     type = SquareType.BONUS_WORD_2X;
101                 } else if (c == 'L') {
102                     type = SquareType.BONUS_LETTER_3X;
103                 } else {
104                     type = SquareType.BONUS_WORD_3X;
105                 }
106                 board[y][x] = new Square(type);
107             }
108         }
109     }
110 }
