6   package fi.luupanu.skrapple.ui.components.dialogs;
7   
8   import fi.luupanu.skrapple.constants.Announcement;
9   import fi.luupanu.skrapple.domain.Player;
10  import fi.luupanu.skrapple.logic.SkrappleGame;
11  import fi.luupanu.skrapple.ui.components.panels.GameScreen;
12  import fi.luupanu.skrapple.utils.Announcer;
13  import java.util.List;
14  import javax.swing.JFrame;
15  import javax.swing.JOptionPane;
16  
17  /**
18   * A game over dialog. Returns to main menu screen when closed.
19   *
20   * @author panu
21   */
22  public class GameOverDialog {
23  
24      private final JFrame frame;
25      private final SkrappleGame s;
26      private final Announcer announcer;
27      private final GameScreen gameScreen;
28  
29      /**
30       * Creates a new GameOverDialog.
31       *
32       * @param a the announcer
33       * @param gameScreen the game screen
34       * @param s SkrappleGame
35       * @param frame the frame to be displayed on
36       */
37      public GameOverDialog(Announcer a, GameScreen gameScreen,
38              SkrappleGame s, JFrame frame) {
39          this.announcer = a;
40          this.frame = frame;
41          this.gameScreen = gameScreen;
42          this.s = s;
43      }
44  
45      /**
46       * Show the game over dialog.
47       */
48      public void showDialog() {
49          List<Player> players = s.doFinalScoring();
50          String message = generateGameOverMessage(players);
51          gameScreen.update(announcer.announce(Announcement.GAME_OVER_MESSAGE));
52          int response = JOptionPane.showConfirmDialog(frame, message, "Game Over :-(",
53                  JOptionPane.OK_CANCEL_OPTION, JOptionPane.WARNING_MESSAGE);
54          gameScreen.showMainMenu();
55      }
56  
57      private String generateGameOverMessage(List<Player> players) {
58          StringBuilder sb = new StringBuilder();
59          sb.append("<html><b>Game over<b><html>!\n\n");
60          sb.append("Final score:\n\n");
61  
62          for (Player p : players) {
63              sb.append(p.getPlayerName()).append(": ");
64              if (p.isResigned()) {
65                  sb.append("RESIGNED");
66              } else {
67                  sb.append(p.getPlayerPoints()).append(" points");
68              }
69              sb.append("\n");
70          }
71          return sb.toString();
72      }
73  }
