6   package fi.luupanu.skrapple.ui.components;
7   
8   import fi.luupanu.skrapple.constants.SkrappleImageIcon;
9   import fi.luupanu.skrapple.domain.Letter;
10  import java.awt.BorderLayout;
11  import java.awt.Color;
12  import java.awt.Font;
13  import javax.swing.JButton;
14  import javax.swing.border.Border;
15  import javax.swing.border.LineBorder;
16  
17  /**
18   * A custom JButton class that optionally contains a Letter. Basically this
19   * class is used to handle rack letter tiles while BoardTile is a subclass for
20   * board letter tiles.
21   *
22   * @author panu
23   */
24  public class LetterTile extends JButton {
25  
26      final Border greenThickBorder = new LineBorder(Color.GREEN, 3);
27      final Border whiteBorder = new LineBorder(Color.WHITE);
28  
29      private Letter letter;
30  
31      /**
32       * Creates a new LetterTile.
33       */
34      public LetterTile() {
35          createJButtonLetter();
36      }
37  
38      /**
39       * Sets a letter to this JButton, sets the correct text (the letter and the
40       * points as a Unicode subscript).
41       *
42       * @param let the letter to be set
43       */
44      public void setLetter(Letter let) {
45          this.letter = let;
46          if (let != null) {
47              this.setText(let.toString() + " " + let.getPointsAsUnicodeSubScript());
48          } else {
49              this.setText("");
50          }
51      }
52  
53      public Letter getLetter() {
54          return letter;
55      }
56  
57      /**
58       * Paints a letter tile.
59       *
60       * @param selected if true, paints a green thick border, otherwise a white
61       * border
62       */
63      public void paintLetterTile(boolean selected) {
64          if (getLetter() != null) {
65              if (selected) {
66                  setBorder(greenThickBorder);
67              } else {
68                  setBorder(whiteBorder);
69              }
70              setIcon(SkrappleImageIcon.LETTER_TILE.getIcon());
71          }
72      }
73  
74      private void createJButtonLetter() {
75          setBorder(new LineBorder(Color.WHITE));
76          setFont(new Font("serif", Font.BOLD, 20));
77          setLayout(new BorderLayout());
78          setHorizontalTextPosition(JButton.CENTER);
79          setVerticalTextPosition(JButton.CENTER);
80          setForeground(new Color(40, 0, 0));
81      }
82  }
