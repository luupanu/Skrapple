6   package fi.luupanu.skrapple.ui.components.panels;
7   
8   import fi.luupanu.skrapple.constants.SkrappleImageIcon;
9   import fi.luupanu.skrapple.filemanager.FileReader;
10  import fi.luupanu.skrapple.logic.SkrappleGame;
11  import fi.luupanu.skrapple.ui.components.LetterTile;
12  import java.awt.Dimension;
13  import java.awt.FlowLayout;
14  import java.awt.Graphics;
15  import java.awt.Image;
16  import java.awt.event.ActionListener;
17  import java.io.IOException;
18  import java.util.logging.Level;
19  import java.util.logging.Logger;
20  import javax.swing.Box;
21  import javax.swing.JPanel;
22  
23  /**
24   * A custom JPanel for the area to display rack letters on.
25   *
26   * @author panu
27   */
28  public class RackPanel extends JPanel {
29  
30      private Image image;
31      private final LetterTile[] rackLetters;
32      private final SkrappleGame s;
33  
34      /**
35       * Creates a new RackPanel.
36       *
37       * @param s SkrappleGame
38       * @param rackLetters the rack letters to be shown on the panel
39       * @param letterActionListener a actionlistener for the lettertiles
40       */
41      public RackPanel(SkrappleGame s,
42              LetterTile[] rackLetters, ActionListener letterActionListener) {
43          this.rackLetters = rackLetters;
44          this.s = s;
45          loadImage();
46          createRackPanel(letterActionListener);
47      }
48  
49      /**
50       * Updates the rack panel to show the current player's rack letters.
51       */
52      public void update() {
53          for (int x = 0; x < rackLetters.length; x++) {
54              rackLetters[x].setLetter(s.getGame().getCurrentPlayer().
55                      getPlayerRack().getContents().get(x));
56          }
57      }
58  
59      /**
60       * Hides all the letter tiles so when switching turns the previous player
61       * doesn't see them.
62       *
63       * @param bool true = visible, false = invisible
64       */
65      public void setLetterTilesVisible(boolean bool) {
66          for (LetterTile tile : rackLetters) {
67              tile.setVisible(bool);
68          }
69      }
70  
71      /**
72       * This method disables or enables all letter tiles.
73       *
74       * @param bool true to enable letter tiles, false to disable
75       */
76      public void setLetterTilesEnabled(boolean bool) {
77          for (LetterTile tile : rackLetters) {
78              tile.setDisabledIcon(tile.getIcon());
79              tile.setEnabled(bool);
80          }
81      }
82  
83      @Override
84      protected void paintComponent(Graphics g) {
85          super.paintComponent(g);
86          drawTiled(g);
87      }
88  
89      private void drawTiled(Graphics g) {
90          Dimension d = getSize();
91          int width = image.getWidth(null);
92          int height = image.getHeight(null);
93  
94          for (int x = 0; x < d.width; x += width) {
95              for (int y = 0; y < d.height; y += height) {
96                  g.drawImage(image, x, y, null, null);
97              }
98          }
99      }
100 
101     private void loadImage() {
102         try {
103             this.image = new FileReader().readImage("images/rackbg.png");
104         } catch (IOException ex) {
105             Logger.getLogger(RackPanel.class.getName()).log(Level.SEVERE, null, ex);
106         }
107     }
108 
109     private void createRackPanel(ActionListener letterActionListener) {
110         setLayout(new FlowLayout(FlowLayout.CENTER, 10, 0));
111 
112         // add invisible rigid width to balance the Undo Move button
113         add(Box.createRigidArea(new Dimension(115, 0)));
114 
115         // add letter tiles
116         for (int x = 0; x < rackLetters.length; x++) {
117             LetterTile b = new LetterTile();
118             b.setIcon(SkrappleImageIcon.LETTER_TILE.getIcon());
119 
120             b.addActionListener(letterActionListener);
121 
122             rackLetters[x] = b;
123             add(rackLetters[x]);
124         }
125         // add some rigid height
126         add(Box.createRigidArea(new Dimension(0, 78)));
127     }
128 }
