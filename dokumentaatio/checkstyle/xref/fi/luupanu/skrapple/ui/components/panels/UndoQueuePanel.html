6   package fi.luupanu.skrapple.ui.components.panels;
7   
8   import fi.luupanu.skrapple.logic.SkrappleGame;
9   import fi.luupanu.skrapple.ui.listeners.UndoLetterQueueActionListener;
10  import java.awt.CardLayout;
11  import javax.swing.JButton;
12  import javax.swing.JPanel;
13  
14  /**
15   * A panel that displays the undo queue button using CardLayout.
16   *
17   * @author panu
18   */
19  public class UndoQueuePanel extends JPanel {
20  
21      private JButton undoQueueButton;
22      private CardLayout cards;
23      private final SkrappleGame s;
24      private final GameScreen gameScreen;
25  
26      /**
27       * Creates a new UndoQueuePanel.
28       * @param s the SkrappleGame
29       * @param gameScreen the game screen to be added to
30       */
31      public UndoQueuePanel(SkrappleGame s, GameScreen gameScreen) {
32          this.gameScreen = gameScreen;
33          this.s = s;
34          createUndoQueuePanel();
35      }
36  
37      @Override
38      public void setVisible(boolean bool) {
39          String visibility = "visible";
40          if (!bool) {
41              visibility = "invisible";
42          }
43          cards.show(this, visibility);
44      }
45  
46      private void createUndoQueuePanel() {
47          cards = new CardLayout();
48          setLayout(cards);
49  
50          // set transparent background
51          setOpaque(false);
52  
53          // add undo queue button
54          undoQueueButton = new JButton("Undo move");
55          undoQueueButton.setFocusable(false);
56          undoQueueButton.addActionListener(new UndoLetterQueueActionListener(s,
57                  gameScreen, undoQueueButton));
58  
59          /*  an empty panel and CardLayout are used instead of setVisible to keep
60              the layout from jumping around when the button visibility is being 
61              toggled */
62          JPanel invisible = new JPanel();
63  
64          // set transparent background
65          invisible.setOpaque(false);
66  
67          // add visible & invisible states to the CardLayout
68          cards.addLayoutComponent(undoQueueButton, "visible");
69          cards.addLayoutComponent(invisible, "invisible");
70  
71          // add components to the panel
72          add(undoQueueButton);
73          add(invisible);
74  
75          // start as invisible
76          setVisible(false);
77      }
78  }
