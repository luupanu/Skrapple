6   package fi.luupanu.skrapple.ui.listeners;
7   
8   import fi.luupanu.skrapple.domain.Dictionary;
9   import fi.luupanu.skrapple.domain.Player;
10  import fi.luupanu.skrapple.logic.SkrappleGame;
11  import fi.luupanu.skrapple.ui.SkrappleGUI;
12  import fi.luupanu.skrapple.ui.components.PlayerTextField;
13  import fi.luupanu.skrapple.ui.components.panels.GameScreen;
14  import java.awt.event.ActionEvent;
15  import java.awt.event.ActionListener;
16  import java.io.IOException;
17  import java.util.ArrayList;
18  import java.util.List;
19  import java.util.logging.Level;
20  import java.util.logging.Logger;
21  import javax.swing.JFrame;
22  import javax.swing.JOptionPane;
23  
24  /**
25   * An action listener to listen to the New Game button on the main menu screen.
26   *
27   * @author panu
28   */
29  public class NewGameActionListener implements ActionListener {
30  
31      private final List<PlayerTextField> playerTextFields;
32      private final JFrame frame;
33      private final SkrappleGUI gui;
34  
35      /**
36       * Creates a new NewGameActionListener.
37       *
38       * @param gui the gui
39       * @param frame the frame to display dialogs on
40       * @param playerTextFields the player text fields
41       */
42      public NewGameActionListener(SkrappleGUI gui, JFrame frame,
43              List<PlayerTextField> playerTextFields) {
44          this.playerTextFields = playerTextFields;
45          this.frame = frame;
46          this.gui = gui;
47      }
48  
49      @Override
50      public void actionPerformed(ActionEvent e) {
51          int n = countNewPlayers();
52          if (n < 2) {
53              JOptionPane.showMessageDialog(frame,
54                      "We need at least 2 players to start ^_^", "Nope!",
55                      JOptionPane.ERROR_MESSAGE);
56          } else {
57              createNewSkrappleGame();
58          }
59      }
60  
61      private void createNewSkrappleGame() {
62          List<Player> players = new ArrayList<>();
63          for (PlayerTextField ptf : playerTextFields) {
64              if (!ptf.getText().isEmpty()) {
65                  players.add(new Player(ptf.getText()));
66              } else {
67                  players.add(null);
68              }
69          }
70          try {
71              SkrappleGame s = new SkrappleGame(players.get(0), players.get(1),
72                      players.get(2), players.get(3),
73                      new Dictionary("kotus-wordlist-fi"));
74              gui.newGameScreen(new GameScreen(gui, s, frame));
75              frame.pack();
76          } catch (IOException ex) {
77              Logger.getLogger(NewGameActionListener.class.getName()).log(Level.SEVERE, null, ex);
78          }
79      }
80  
81      private int countNewPlayers() {
82          int n = 0;
83          for (PlayerTextField ptf : playerTextFields) {
84              if (!ptf.getText().isEmpty()) {
85                  n++;
86              }
87          }
88          return n;
89      }
90  }
