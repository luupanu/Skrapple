143     private void makeBoardPanel(JPanel boardPanel) {
144         // set layout
145         GridBagLayout layout = new GridBagLayout();
146         GridBagConstraints gbc = new GridBagConstraints();
147         boardPanel.setLayout(layout);
148 
149         // set border
150         boardPanel.setBorder(BorderFactory.createBevelBorder(
151                 BevelBorder.LOWERED, Color.LIGHT_GRAY, Color.DARK_GRAY));
152 
153         // create board
154         JPanel board = new JPanel(new GridLayout(0, BOARD_MAX_SIZE));
155         makeBoard(board);
156 
157         // create rack
158         rackPanel = new RackPanel(new FlowLayout(FlowLayout.CENTER, 10, 0));
159         makeRack(rackPanel);
160 
161         // create undo queue button & panel
162         makeUndoQueuePanel(rackPanel);
163 
164         gbc.fill = GridBagConstraints.BOTH;
165         gbc.weightx = 1;
166         gbc.weighty = 0.9375;
167         boardPanel.add(board, gbc);
168 
169         gbc.gridy = 1;
170         gbc.weighty = 0.0625;
171         boardPanel.add(rackPanel, gbc);
172     }

174     private void makeBoard(JPanel board) {
175         // set background
176         board.setBackground(new Color(154, 146, 125));
177 
178         Insets buttonMargin = new Insets(0, 0, 0, 0);
179         String[] layout = s.getGame().getBoard().getLayout();
180         for (int y = 0; y < boardSquares.length; y++) {
181             for (int x = 0; x < boardSquares[y].length; x++) {
182                 LetterTile b = new LetterTile(true);
183                 b.setCoord(new Coord(x, y));
184                 b.setMargin(buttonMargin);
185 
186                 b.paintBoardIcon(layout);
187 
188                 b.addActionListener(letterListener);
189 
190                 boardSquares[x][y] = b;
191                 board.add(boardSquares[x][y]);
192             }
193         }
194     }

196     private void makeRack(RackPanel rackPanel) {
197         // add invisible area to balance Undo Move button
198         rackPanel.add(Box.createRigidArea(new Dimension(115, 0)));
199         for (int x = 0; x < rackLetters.length; x++) {
200             LetterTile b = new LetterTile(false);
201             b.setIcon(SkrappleImageIcon.LETTER_TILE.getIcon());
202             //b.setAlignmentY(Component.CENTER_ALIGNMENT);
203 
204             b.addActionListener(letterListener);
205 
206             rackLetters[x] = b;
207             rackPanel.add(rackLetters[x]);
208         }
209         rackPanel.add(Box.createRigidArea(new Dimension(0, 78)));
210     }

212     private void makeUndoQueuePanel(RackPanel rackPanel) {
213         // set layout
214         undoQueuePanel = new JPanel();
215         cards = new CardLayout();
216         undoQueuePanel.setLayout(cards);
217 
218         // set transparent background
219         undoQueuePanel.setBackground(new Color(255, 255, 255, 0));
220 
221         // add undo queue button
222         undoQueueButton = new JButton("Undo move");
223         undoQueueButton.setFocusable(false);
224         undoQueueButton.addActionListener(new UndoLetterQueueActionListener(s,
225                 this, rackPanel, undoQueueButton));
226 
227         /*  an empty panel and CardLayout are used instead of setVisible to keep
228             the layout from jumping around when the button visibility is being 
229             toggled */
230         JPanel invisible = new JPanel();
231 
232         // set transparent background
233         invisible.setBackground(new Color(255, 255, 255, 0));
234 
235         // add visible & invisible states to the CardLayout
236         cards.addLayoutComponent(undoQueueButton, "visible");
237         cards.addLayoutComponent(invisible, "invisible");
238 
239         // add components to the panel
240         undoQueuePanel.add(undoQueueButton);
241         undoQueuePanel.add(invisible);
242 
243         // start as invisible
244         cards.show(undoQueuePanel, "invisible");
245 
246         // add to rackPanel
247         rackPanel.add(undoQueuePanel);
248     }

340     private void makeInfoBox(JPanel sidePanelInfoBox) {
341         // set background & border
342         sidePanelInfoBox.setBackground(Color.LIGHT_GRAY);
343         sidePanelInfoBox.setBorder(BorderFactory.createLoweredBevelBorder());
344 
345         // create info field text area
346         infoField = new JTextArea(10, 18);
347         infoField.setEditable(false);
348         infoField.setLineWrap(true);
349         infoField.setWrapStyleWord(true);
350         infoField.setFont(normal);
351 
352         // create vertical scrollpane
353         JScrollPane sp = new JScrollPane(infoField,
354                 JScrollPane.VERTICAL_SCROLLBAR_ALWAYS,
355                 JScrollPane.HORIZONTAL_SCROLLBAR_NEVER);
356 
357         // auto-scroll to bottom
358         DefaultCaret caret = (DefaultCaret) infoField.getCaret();
359         caret.setUpdatePolicy(DefaultCaret.ALWAYS_UPDATE);
360 
361         // add text area with scrollpane
362         sidePanelInfoBox.add(sp);
363     }

365     private void makeSidePanelButtons(JPanel sidePanelButtons) {
366         // set layout
367         GridLayout layout = new GridLayout(4, 1);
368         sidePanelButtons.setLayout(layout);
369 
370         // create buttons
371         moveButton = new JButton("Confirm move");
372         endTurnButton = new JButton("End turn");
373         exchangeButton = new JButton("Exchange letters");
374         resignButton = new JButton("Resign");
375 
376         // add ActionListeners
377         moveButton.addActionListener(new MoveActionListener(announcer, this, s, frame,
378                 moveButton, exchangeButton));
379         endTurnButton.addActionListener(new EndTurnActionListener(announcer, this, s,
380                 frame, endTurnButton, moveButton, exchangeButton));
381         exchangeButton.addActionListener(letterListener);
382         resignButton.addActionListener(new ResignActionListener(s, frame, resignButton,
383                 playerOneName, playerTwoName, playerOnePoints, playerTwoPoints));
384 
385         // add buttons
386         sidePanelButtons.add(moveButton);
387         sidePanelButtons.add(endTurnButton);
388         sidePanelButtons.add(exchangeButton);
389         sidePanelButtons.add(resignButton);
390     }

392     public void updateBoardLettersColour() {
393         for (LetterTile[] jbls : boardSquares) {
394             for (LetterTile jbl : jbls) {
395                 if (jbl.getLetter() != null) {
396                     jbl.setIcon(SkrappleImageIcon.LETTER_TILE.getIcon());
397                 }
398             }
399         }

407     public void updateHangingLetterTiles() {
408         for (LetterTile[] tiles : boardSquares) {
409             for (LetterTile tile : tiles) {
410                 if (tile.getIcon() == SkrappleImageIcon.LETTER_TILE_SELECTED.getIcon()) {
411                     tile.paintBoardIcon(s.getGame().getBoard().getLayout());
412                 }
413             }
414         }
415     }

417     public void setLetterTilesEnabled(boolean bool) {
418         for (LetterTile jbl : rackLetters) {
419             jbl.setDisabledIcon(jbl.getIcon());
420             jbl.setEnabled(bool);
421         }
422         for (LetterTile[] jbls : boardSquares) {
423             for (LetterTile jbl : jbls) {
424                 jbl.setDisabledIcon(jbl.getIcon());
425                 jbl.setEnabled(bool);
426             }
427         }
428     }

430     public void updatePlayerInfo() {
431         if (s.getGame().getTurn()) {
432             for (Component c : playerOne.getComponents()) {
433                 c.setFont(bolded);
434             }
435             for (Component c : playerTwo.getComponents()) {
436                 c.setFont(normal);
437             }
438             playerOne.setBorder(BorderFactory.createEtchedBorder());
439             playerTwo.setBorder(null);
440         } else {
441             for (Component c : playerOne.getComponents()) {
442                 c.setFont(normal);
443             }
444             for (Component c : playerTwo.getComponents()) {
445                 c.setFont(bolded);
446             }
447             playerOne.setBorder(null);
448             playerTwo.setBorder(BorderFactory.createEtchedBorder());
449         }
450     }
