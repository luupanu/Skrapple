6   package fi.luupanu.skrapple.ui;
7   
8   import fi.luupanu.skrapple.ui.components.panels.GameScreen;
9   import fi.luupanu.skrapple.ui.components.panels.MainMenuScreen;
10  import java.awt.CardLayout;
11  import java.awt.Color;
12  import java.awt.Dimension;
13  import javax.swing.JFrame;
14  import javax.swing.SwingUtilities;
15  import javax.swing.UIManager;
16  import javax.swing.WindowConstants;
17  
18  /**
19   * A giant mess of a graphical UI.
20   *
21   * @author panu
22   */
23  public class SkrappleGUI implements Runnable {
24  
25      private JFrame frame;
26      private CardLayout cards;
27      private MainMenuScreen mainMenuScreen;
28  
29      /**
30       * Creates a new SkrappleGUI.
31       */
32      public SkrappleGUI() {
33          UIManager.put("Button.disabledText", new Color(130, 0, 0));
34      }
35  
36      @Override
37      public void run() {
38          frame = new JFrame("Skrapple");
39          frame.setMinimumSize(new Dimension(800, 600));
40          frame.setPreferredSize(new Dimension(993, 802));
41          frame.setDefaultCloseOperation(WindowConstants.EXIT_ON_CLOSE);
42  
43          addComponents();
44  
45          frame.pack();
46          mainMenuScreen.requestFocusForQuitButton();
47          frame.setVisible(true);
48      }
49  
50      /**
51       * Shows a new game screen.
52       *
53       * @param gameScreen the game screen to be shown
54       */
55      public void newGameScreen(GameScreen gameScreen) {
56          frame.getContentPane().add(gameScreen);
57          cards.addLayoutComponent(gameScreen, "game");
58          show("game");
59      }
60  
61      private void addComponents() {
62          mainMenuScreen = new MainMenuScreen(this, frame);
63  
64          cards = new CardLayout();
65          frame.getContentPane().setLayout(cards);
66  
67          cards.addLayoutComponent(mainMenuScreen, "menu");
68  
69          frame.getContentPane().add(mainMenuScreen);
70  
71          show("menu");
72      }
73  
74      /**
75       * Use CardLayout to show this screen.
76       *
77       * @param screen the string identifier of the screen to be shown
78       */
79      public void show(String screen) {
80          cards.show(frame.getContentPane(), screen);
81      }
82  
83      /**
84       * Main method of the whole program.
85       * @param args args 
86       */
87      public static void main(String[] args) {
88          SwingUtilities.invokeLater(new SkrappleGUI());
89      }
90  }
