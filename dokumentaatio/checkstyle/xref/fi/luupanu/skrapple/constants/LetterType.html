6   package fi.luupanu.skrapple.constants;
7   
8   /**
9    * LetterType contains the different types of letters available in the game.
10   * 
11   * @author panu
12   */
13  public enum LetterType {
14      LETTER_A("A", 1),
15      LETTER_B("B", 8),
16      LETTER_C("C", 10),
17      LETTER_D("D", 7),
18      LETTER_E("E", 1),
19      LETTER_F("F", 8),
20      LETTER_G("G", 8),
21      LETTER_H("H", 4),
22      LETTER_I("I", 1),
23      LETTER_J("J", 4),
24      LETTER_K("K", 2),
25      LETTER_L("L", 2),
26      LETTER_M("M", 3),
27      LETTER_N("N", 1),
28      LETTER_O("O", 2),
29      LETTER_P("P", 4), // letter Q does not exist in the Finnish version
30      LETTER_R("R", 4),
31      LETTER_S("S", 1),
32      LETTER_T("T", 1),
33      LETTER_U("U", 3),
34      LETTER_V("V", 4),
35      LETTER_W("W", 8),
36      LETTER_Y("Y", 4), // nor Z
37      LETTER_AE("Ä", 2),
38      LETTER_OE("Ö", 7),
39      LETTER_WILD("*", 0);
40      private final int points;
41      private final String name;
42  
43      private LetterType(String name, int points) {
44          this.name = name;
45          this.points = points;
46      }
47  
48      public String getName() {
49          return this.name;
50      }
51  
52      public int getPoints() {
53          return this.points;
54      }
55  }
