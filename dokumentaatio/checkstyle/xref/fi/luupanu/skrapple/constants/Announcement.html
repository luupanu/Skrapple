6   package fi.luupanu.skrapple.constants;
7   
8   /**
9    * Announcements are generic strings with replaceable elements that are used to
10   * announce game events to the player.
11   *
12   * @author panu
13   */
14  public enum Announcement implements Message {
15      WELCOME_MESSAGE("Welcome to Skrapple!"),
16      TURN_START_MESSAGE("\nIt's <PLAYER>'s turn."),
17      PLAYER_RESIGNED_MESSAGE("<PLAYER> resigned!"),
18      PLAYER_EXCHANGED_MESSAGE("<PLAYER> exchanged their letters."),
19      LETTERQUEUE_EMPTY_MESSAGE("You didn't place any letters, silly you! ^_^"),
20      LETTERQUEUE_NOT_VALID_MESSAGE("Can't make an invalid move, now can we? ^_^"),
21      WORD_NOT_VALID_MESSAGE("<WORD>? That's not a word! ^_^"),
22      WORD_SCORE_MESSAGE("Word <WORD> scored <WORDPOINTS> points."),
23      BONUS_SCORE_MESSAGE("You used all seven letter tiles! 50 bonus points scored."),
24      REFILL_RACK_MESSAGE("Added new letters <LETTERS> to the rack."),
25      LETTERBAG_EMPTY_MESSAGE("The letter bag is empty!"),
26      EXCHANGE_FAILED_MESSAGE("The letter bag has less letters than you're trying to exchange!"),
27      EXCHANGE_CANNOT_START_MESSAGE("Are you trying to move and exchange at the same time? That's nitwitted! ^_^"),
28      RACK_EMPTY_MESSAGE("<PLAYER> has no letters! The game is over!"),
29      GAME_OVER_MESSAGE("\nGood game! ^_^");
30  
31      private String message;
32  
33      private Announcement(String message) {
34          setMessage(message);
35      }
36  
37      private void setMessage(String message) {
38          this.message = message;
39      }
40  
41      @Override
42      public String getMessage() {
43          return message;
44      }
45  }
